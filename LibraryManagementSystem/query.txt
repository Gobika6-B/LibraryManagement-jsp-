CREATE TABLE LIBRARY_TB (
    RECORDID VARCHAR2(12 BYTE) PRIMARY KEY,
    BOOKTITLE VARCHAR2(60 BYTE) NOT NULL,
    AUTHOR VARCHAR2(40 BYTE) NOT NULL,
    CATEGORY VARCHAR2(30 BYTE),
    PUBLISHED_YEAR NUMBER,
    ADDED_DATE DATE NOT NULL,
    REMARKS VARCHAR2(100 BYTE)
);

CREATE SEQUENCE LIBRARY_SEQ
START WITH 10
MAXVALUE 99
INCREMENT BY 1;
